      div.classList.add("section", "recommendation");
      div.innerHTML = `<strong>Recommendation:</strong> ${rec}`;
      resultsContainer.appendChild(div);
    });

    // Render issues
    aiResponse.issues.forEach(issue => {
      const div = document.createElement("div");

      resultsContainer.appendChild(div);
    });
